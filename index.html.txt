```html
<!DOCTYPE html>
<html lang="ja">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0, maximum-scale=1.0, user-scalable=no">
    <title>超ダークモード静音カメラ</title>
    <!-- Tailwind CSS for styling -->
    <script src="https://cdn.tailwindcss.com"></script>
    <style>
        /* タップ時のハイライトなどを無効化して誤操作を防ぐ */
        * {
            -webkit-tap-highlight-color: transparent;
            user-select: none;
        }
        body, html {
            margin: 0;
            padding: 0;
            width: 100%;
            height: 100%;
            background-color: #000000;
            overflow: hidden;
            color: #111111; /* 画面をほぼ真っ黒にするため、文字も極めて暗く配置 */
            font-family: -apple-system, BlinkMacSystemFont, "Segoe UI", Roboto, sans-serif;
        }
        /* カメラのプレビューを非表示（または超極小化）にする */
        #previewVideo {
            position: absolute;
            top: 0;
            left: 0;
            width: 1px;
            height: 1px;
            opacity: 0.01;
            pointer-events: none;
        }
    </style>
</head>
<body class="flex flex-col justify-between items-center p-6">

    <!-- カメラプレビュー（画面上はほぼ見えないように隠します） -->
    <video id="previewVideo" autoplay playsinline muted></video>

    <!-- ヘッダー：極限まで視認性を下げたステータス表示 -->
    <div class="w-full flex justify-between items-center opacity-10 text-[10px]">
        <span id="statusText">SYSTEM: READY</span>
        <span id="modeText">MODE: PHOTO</span>
    </div>

    <!-- メイン操作エリア（画面全体のどこをタップしても動作するようにします） -->
    <div id="gestureArea" class="flex-1 w-full flex flex-col justify-center items-center cursor-pointer relative">
        <!-- 画面が真っ黒だと動いているか分からないので、ごく微かにガイドラインを配置（設定で消去可能） -->
        <div id="guideOverlay" class="text-center transition-opacity duration-300 opacity-20">
            <p class="text-xs text-neutral-400 mb-2">【操作方法】</p>
            <p class="text-[11px] text-neutral-500">画面タップ：写真を撮影</p>
            <p class="text-[11px] text-neutral-500">画面長押し：動画の録画 開始／停止</p>
            <p class="text-[11px] text-neutral-500">2本指タップ：プレビュー表示のON/OFF</p>
        </div>
        
        <!-- 録画中の微かなインジケーター（ほぼ見えない明るさの赤ドット） -->
        <div id="recordingIndicator" class="absolute top-4 right-4 w-2 h-2 rounded-full bg-red-600 hidden opacity-30"></div>
    </div>

    <!-- フッター：超ダークモード用コントロール（極めて薄い表示） -->
    <div class="w-full flex justify-around items-center opacity-10 text-[11px] border-t border-neutral-900 pt-4">
        <button id="togglePreviewBtn" class="px-3 py-1 rounded bg-neutral-900 text-neutral-400">プレビュー切替</button>
        <button id="switchModeBtn" class="px-3 py-1 rounded bg-neutral-900 text-neutral-400">写真 ⇄ 動画</button>
        <button id="hideGuiBtn" class="px-3 py-1 rounded bg-neutral-900 text-neutral-400">案内を非表示</button>
    </div>

    <!-- 簡易メッセージトースト（画面に大きなUIを出さないよう、画面最下部に薄く表示） -->
    <div id="toast" class="fixed bottom-16 left-1/2 transform -translate-x-1/2 bg-neutral-950 text-neutral-500 text-[10px] px-3 py-1 rounded-full opacity-0 transition-opacity duration-300 pointer-events-none"></div>

    <script>
        // グローバル変数定義
        const video = document.getElementById('previewVideo');
        const gestureArea = document.getElementById('gestureArea');
        const statusText = document.getElementById('statusText');
        const modeText = document.getElementById('modeText');
        const togglePreviewBtn = document.getElementById('togglePreviewBtn');
        const switchModeBtn = document.getElementById('switchModeBtn');
        const hideGuiBtn = document.getElementById('hideGuiBtn');
        const guideOverlay = document.getElementById('guideOverlay');
        const recordingIndicator = document.getElementById('recordingIndicator');
        const toast = document.getElementById('toast');

        let mediaStream = null;
        let mediaRecorder = null;
        let recordedChunks = [];
        let isRecording = false;
        let currentMode = 'PHOTO'; // PHOTO or VIDEO
        let isPreviewVisible = false;
        let touchTimer = null;
        let isLongTouch = false;

        // トーストメッセージ表示
        function showToast(message) {
            toast.textContent = message;
            toast.classList.remove('opacity-0');
            toast.classList.add('opacity-30'); // 薄く表示
            setTimeout(() => {
                toast.classList.remove('opacity-30');
                toast.classList.add('opacity-0');
            }, 1500);
        }

        // カメラの初期化
        async function initCamera() {
            try {
                // 背面カメラ（environment）を優先して取得
                const constraints = {
                    video: {
                        facingMode: 'environment',
                        width: { ideal: 1920 },
                        height: { ideal: 1080 }
                    },
                    audio: true // 動画録音用に音声も許可
                };

                mediaStream = await navigator.mediaDevices.getUserMedia(constraints);
                video.srcObject = mediaStream;
                statusText.textContent = "SYSTEM: ACTIVE";
                showToast("カメラ起動完了");
            } catch (err) {
                console.error("カメラの起動に失敗しました:", err);
                statusText.textContent = "SYSTEM: ERROR";
                showToast("カメラの権限を許可してください");
            }
        }

        // 写真撮影処理（Canvasに描画してダウンロード）
        function takePhoto() {
            if (!mediaStream) return;

            const canvas = document.createElement('canvas');
            canvas.width = video.videoWidth || 1280;
            canvas.height = video.videoHeight || 720;
            
            const ctx = canvas.getContext('2d');
            // 映像をCanvasに描画
            ctx.drawImage(video, 0, 0, canvas.width, canvas.height);
            
            // データURLに変換して即座に保存
            try {
                const dataUrl = canvas.toDataURL('image/jpeg');
                const link = document.createElement('a');
                link.download = `photo_${Date.now()}.jpg`;
                link.href = dataUrl;
                link.click();
                
                // 撮影完了のフィードバック（画面を一瞬だけごく微かに点滅）
                document.body.style.backgroundColor = '#111111';
                setTimeout(() => {
                    document.body.style.backgroundColor = '#000000';
                }, 80);

                showToast("写真を保存しました");
            } catch (err) {
                console.error("写真の保存に失敗しました:", err);
                showToast("保存失敗");
            }
        }

        // 動画録画の開始
        function startRecording() {
            if (!mediaStream || isRecording) return;

            recordedChunks = [];
            // MIMEタイプの判定（ブラウザの対応状況に合わせる）
            let options = { mimeType: 'video/webm;codecs=vp9,opus' };
            if (!MediaRecorder.isTypeSupported(options.mimeType)) {
                options = { mimeType: 'video/webm;codecs=vp8,opus' };
                if (!MediaRecorder.isTypeSupported(options.mimeType)) {
                    options = { mimeType: 'video/mp4' };
                }
            }

            try {
                mediaRecorder = new MediaRecorder(mediaStream, options);
            } catch (e) {
                mediaRecorder = new MediaRecorder(mediaStream);
            }

            mediaRecorder.ondataavailable = (event) => {
                if (event.data && event.data.size > 0) {
                    recordedChunks.push(event.data);
                }
            };

            mediaRecorder.onstop = () => {
                const blob = new Blob(recordedChunks, { type: mediaRecorder.mimeType || 'video/webm' });
                const url = URL.createObjectURL(blob);
                const link = document.createElement('a');
                link.download = `video_${Date.now()}.${blob.type.includes('mp4') ? 'mp4' : 'webm'}`;
                link.href = url;
                link.click();
                showToast("ビデオを保存しました");
            };

            mediaRecorder.start();
            isRecording = true;
            recordingIndicator.classList.remove('hidden');
            statusText.textContent = "SYSTEM: RECORDING";
            showToast("録画開始");
        }

        // 動画録画の停止
        function stopRecording() {
            if (!mediaRecorder || !isRecording) return;
            mediaRecorder.stop();
            isRecording = false;
            recordingIndicator.classList.add('hidden');
            statusText.textContent = "SYSTEM: ACTIVE";
            showToast("録画停止");
        }

        // プレビューの表示・非表示切り替え
        function togglePreview() {
            if (isPreviewVisible) {
                // 非表示にする（極小化して透過率を下げる）
                video.style.width = '1px';
                video.style.height = '1px';
                video.style.opacity = '0.01';
                video.style.position = 'absolute';
                isPreviewVisible = false;
                showToast("プレビュー非表示");
            } else {
                // 画面全体に表示する（実際の構図確認用）
                video.style.width = '100%';
                video.style.height = '100%';
                video.style.opacity = '0.3'; // 眩しくないように少し暗めにする
                video.style.position = 'absolute';
                video.style.objectFit = 'cover';
                isPreviewVisible = true;
                showToast("プレビュー簡易表示");
            }
        }

        // モード切り替え
        function toggleMode() {
            if (currentMode === 'PHOTO') {
                currentMode = 'VIDEO';
                modeText.textContent = "MODE: VIDEO";
                showToast("動画モード");
            } else {
                if (isRecording) stopRecording();
                currentMode = 'PHOTO';
                modeText.textContent = "MODE: PHOTO";
                showToast("写真モード");
            }
        }

        // --- タッチおよびクリックイベントの処理 ---

        // ジェスチャーエリアでのインタラクション
        gestureArea.addEventListener('touchstart', (e) => {
            // 2本指でのマルチタップを検出
            if (e.touches.length === 2) {
                clearTimeout(touchTimer);
                isLongTouch = false;
                togglePreview();
                return;
            }

            isLongTouch = false;
            // 600ms以上の長押しで「動画録画の開始/停止」
            touchTimer = setTimeout(() => {
                isLongTouch = true;
                if (currentMode === 'VIDEO') {
                    if (!isRecording) {
                        startRecording();
                    } else {
                        stopRecording();
                    }
                } else {
                    showToast("動画モードで長押ししてください");
                }
            }, 600);
        });

        gestureArea.addEventListener('touchend', (e) => {
            clearTimeout(touchTimer);
            if (!isLongTouch && e.touches.length === 0) {
                // シングルタップの処理
                if (currentMode === 'PHOTO') {
                    takePhoto();
                } else {
                    // ビデオモードで通常タップされた場合
                    if (isRecording) {
                        stopRecording();
                    } else {
                        showToast("録画は長押ししてください");
                    }
                }
            }
        });

        // PCブラウザでの動作確認用のマウスクリックイベント
        gestureArea.addEventListener('mousedown', (e) => {
            isLongTouch = false;
            touchTimer = setTimeout(() => {
                isLongTouch = true;
                if (currentMode === 'VIDEO') {
                    if (!isRecording) startRecording();
                    else stopRecording();
                }
            }, 600);
        });

        gestureArea.addEventListener('mouseup', (e) => {
            clearTimeout(touchTimer);
            if (!isLongTouch) {
                if (currentMode === 'PHOTO') {
                    takePhoto();
                }
            }
        });

        // 各種ボタンのイベントリスナー
        togglePreviewBtn.addEventListener('click', (e) => {
            e.stopPropagation(); // ジェスチャーエリアのイベント発火を防ぐ
            togglePreview();
        });

        switchModeBtn.addEventListener('click', (e) => {
            e.stopPropagation();
            toggleMode();
        });

        hideGuiBtn.addEventListener('click', (e) => {
            e.stopPropagation();
            if (guideOverlay.style.opacity === '0') {
                guideOverlay.style.opacity = '0.2';
                hideGuiBtn.textContent = "案内を非表示";
            } else {
                guideOverlay.style.opacity = '0';
                hideGuiBtn.textContent = "案内を表示";
            }
        });

        // ページロード時にカメラを起動
        window.onload = () => {
            initCamera();
        };
    </script>
</body>
</html>

```
